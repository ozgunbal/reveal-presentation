Lifecycle: Mount / Init

React Native


import React, { useEffect, useState } from 'react';
import { View, Text, TouchableOpacity } from 'react-native';

function LifecycleComponent() {
  const [count, setCount] = useState(0);
  const [data, setData] = useState(null);

  useEffect(() => {
    // Mount: runs once when component mounts
    console.log('Component mounted');

    // Async operation returning a Promise
    fetchData().then(result => setData(result));

    return () => {
      // Unmount: cleanup function
      console.log('Component will unmount');
    };
  }, []);

  useEffect(() => {
    // Update: runs when count changes
    console.log('Count updated:', count);
  }, [count]);

  const fetchData = () => {
    return new Promise(resolve => {
      setTimeout(() => resolve('Data loaded'), 1000);
    });
  };

  return (
    <View>
      <Text>Count: {count}</Text>
      <Text>Data: {data}</Text>
      <TouchableOpacity onPress={() => setCount(count + 1)}>
        <Text>Increment</Text>
      </TouchableOpacity>
    </View>
  );
}

Flutter


class LifecycleWidget extends StatefulWidget {
  @override
  _LifecycleWidgetState createState() => _LifecycleWidgetState();
}

class _LifecycleWidgetState extends State<LifecycleWidget> {
  int count = 0;
  String? data;

  @override
  void initState() {
    // Mount: runs once when widget is inserted
    super.initState();
    print('Widget initialized');

    // Async operation returning a Future
    fetchData().then((result) => setState(() => data = result));
  }

  @override
  void didUpdateWidget(LifecycleWidget oldWidget) {
    // Update: runs when widget configuration changes
    super.didUpdateWidget(oldWidget);
    print('Widget updated');
  }

  @override
  void dispose() {
    // Unmount: cleanup when widget is removed
    print('Widget disposed');
    super.dispose();
  }

  Future<String> fetchData() async {
    await Future.delayed(Duration(seconds: 1));
    return 'Data loaded';
  }

  @override
  Widget build(BuildContext context) {
    return Column(
      children: [
        Text('Count: $count'),
        Text('Data: ${data ?? "Loading..."}'),
        ElevatedButton(
          onPressed: () => setState(() => count++),
          child: Text('Increment'),
        ),
      ],
    );
  }
}

React Native Flutter

Arguments: Named Props vs Positional & Named

React Native - Only Named Props


import React from 'react';
import { View, Text } from 'react-native';

function Button({ title, onPress, color, disabled }) {
  return (
    <TouchableOpacity
      onPress={onPress}
      disabled={disabled}
      style={{ backgroundColor: color }}>
      <Text>{title}</Text>
    </TouchableOpacity>
  );
}

function App() {
  return (
    <Button
      title="Click me"
      onPress={() => console.log('Pressed')}
      color="blue"
      disabled={false}
    />
  );
}

Flutter - Positional & Named Arguments


import 'package:flutter/material.dart';

class CustomButton extends StatelessWidget {
  final String title;           // Positional
  final VoidCallback onPressed; // Positional
  final Color? color;           // Named (optional)
  final bool? disabled;         // Named (optional)

  CustomButton(
    this.title,
    this.onPressed,
    {this.color, this.disabled = false}
  );

  @override
  Widget build(BuildContext context) {
    return ElevatedButton(
      onPressed: disabled! ? null : onPressed,
      style: ElevatedButton.styleFrom(
        backgroundColor: color ?? Colors.blue,
      ),
      child: Text(title),
    );
  }
}

class MyApp extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return CustomButton(
      'Click me',
      () => print('Pressed'),
      color: Colors.blue,
      disabled: false,
    );
  }
}
